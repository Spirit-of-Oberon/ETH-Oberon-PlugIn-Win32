MODULE ProfilerExample; (** portable *)
IMPORT Oberon, Out, Texts;
	VAR
		i: INTEGER;
		W: Texts.Writer;

PROCEDURE UseWriter(i: INTEGER);
BEGIN
	Texts.WriteInt(W, i, 3)
END UseWriter;

PROCEDURE UseOut(i: INTEGER);
BEGIN
	Out.Int(i, 3)
END UseOut;

PROCEDURE Do*;
BEGIN
END Do;

BEGIN
	i := 0;
	Texts.OpenWriter(W);
	WHILE i < 100 DO
		UseWriter(i);
		INC(i)
	END;
	Texts.WriteLn(W);
	Texts.Append(Oberon.Log, W.buf);
	
	i := 0;
	WHILE i < 100 DO
		UseOut(i);
		INC(i)
	END;
	Out.Ln;
END ProfilerExample.
